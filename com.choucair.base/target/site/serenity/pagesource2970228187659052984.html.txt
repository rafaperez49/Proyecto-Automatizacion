			return false;
		   }
	   return true;
		}

	function isDateDDMMYYYY(field){
	  if(field.value.length != 10){
	    return false;
	  }else if(field.value.substring(2,3) != "/" || field.value.substring(5,6) != "/"){
	    return false;
	  }else if(isNaN(field.value.substring(0,2)) || isNaN(field.value.substring(3,5))){
	    return false;
	  }
	
	  
	  var day   = field.value.substring(0,2);
	  var month = field.value.substring(3,5);
	  var year  = field.value.substring(6,10);
	     if (isNaN(year) || parseInt(year)&lt; 1800) {
	  	return false;
	  }
	
	  var monthint;
	  if (month.substring(0,1) != '0') {
	    monthint = parseInt(month);
	  }
	  else {
	    monthint = parseInt(month.substring(1,2));
	  }
	  var dayint;
	  if (day.substring(0,1) != '0') {
	    dayint = parseInt(day);
	  }
	  else {
	    dayint = parseInt(day.substring(1,2));
	  }
	  var yearint = parseInt(year);
	
	  
	  
	  var date = new Date(yearint, monthint - 1, dayint);
	
	  
	  if (date.getYear().length == 2) {
	      if ((date.getYear() + 1900) != yearint) {
	          return false;
	      }
	  }
	  
	  
	  if (date.getYear().length == 4) {
	      if (date.getYear() != yearint) {
	          return false;
	      }
	  }
	
	  if (((date.getMonth() + 1) != monthint) ||
	      (date.getDate() != dayint)) {
	      return false;
	  }
	  
	  return true;
	}

	function trimSpaces(modelName) {
		var i = 0;
		while (i &lt; modelName.length &amp;&amp; modelName.charAt(i) == " ") {
			i++;
		}

		if(i &lt; modelName.length) {
			modelName = modelName.substring(i, modelName.length);
		}

		i = modelName.length - 1;
		while (i &gt; 0 &amp;&amp; modelName.charAt(i) == " ") {
			i--;
		}

		if(i &gt; 0) {
			modelName = modelName.substring(0, i+1);
		}
		return modelName;
	}
